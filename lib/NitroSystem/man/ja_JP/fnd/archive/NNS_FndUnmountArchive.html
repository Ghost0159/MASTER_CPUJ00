NNS_FndUnmountArchive

構文

#include <nnsys/fnd.h>
BOOL NNS_FndUnmountArchive(NNSFndArchive* archive);

引数

archive NNSFndArchive構造体へのポインタ

返り値

アンマウントに成功すれば、真を返します。

説明

archiveで指定されたファイルシステムにマウントされているアーカイブをアンマウントします。 


    extern void* archiveBinary;

    NNSFndArchive archive;

    if (NNS_FndMountArchive(&archive, "ARC", archiveBinary))
    {
        ......
        ......

        NNS_FndUnmountArchive(&archive);
    }
