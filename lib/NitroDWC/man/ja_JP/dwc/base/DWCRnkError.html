DWCRnkError

定義


#include <dwc.h>

typedef enum{

    DWC_RNK_SUCCESS = 0,
    DWC_RNK_IN_ERROR,

    DWC_RNK_ERROR_INVALID_PARAMETER,

    DWC_RNK_ERROR_INIT_ALREADYINITIALIZED,
    DWC_RNK_ERROR_INIT_INVALID_INITDATASIZE,
    DWC_RNK_ERROR_INIT_INVALID_INITDATA,
    DWC_RNK_ERROR_INIT_INVALID_USERDATA,

    DWC_RNK_ERROR_PUT_NOTINITIALIZED,
    DWC_RNK_ERROR_PUT_INVALID_KEY,
    DWC_RNK_ERROR_PUT_NOMEMORY,
    
    DWC_RNK_ERROR_GET_NOTINITIALIZED,
    DWC_RNK_ERROR_GET_INVALID_KEY,
    DWC_RNK_ERROR_GET_NOMEMORY,

    DWC_RNK_ERROR_CANCEL_NOTASK,

    DWC_RNK_PROCESS_NOTASK,

    DWC_RNK_ERROR_INVALID_MODE,
    DWC_RNK_ERROR_NOTCOMPLETED,
    DWC_RNK_ERROR_EMPTY_RESPONSE

} DWCRnkError;

説明

汎用ランキングのエラーコードを示す列挙型です。

DWC_RNK_SUCCESS 成功
DWC_RNK_IN_ERROR エラー発生中
DWC_RNK_ERROR_INVALID_PARAMETER 不正なパラメータ
DWC_RNK_ERROR_INIT_ALREADYINITIALIZED DWC_RnkInitialize 初期化済
DWC_RNK_ERROR_INIT_INVALID_INITDATASIZE DWC_RnkInitialize 不正な初期化データサイズ
DWC_RNK_ERROR_INIT_INVALID_INITDATA DWC_RnkInitialize 不正な初期化データ
DWC_RNK_ERROR_INIT_INVALID_USERDATA DWC_RnkInitialize 不正なユーザーデータ
ユーザーデータは有効なGameSpy Profile IDを含む必要があります。一度もログインしていない仮IDを含むデータではこのエラーが発生します。
DWC_RNK_ERROR_PUT_NOTINITIALIZED DWC_RnkPutScoreAsync 初期化されていない
DWC_RNK_ERROR_PUT_INVALID_KEY DWC_RnkPutScoreAsync 暗号化キーが不正
DWC_RNK_ERROR_PUT_NOMEMORY DWC_RnkPutScoreAsync メモリー不足
DWC_RNK_ERROR_GET_NOTINITIALIZED DWC_RnkGetScoreAsync 初期化されていない
DWC_RNK_ERROR_GET_INVALID_KEY DWC_RnkGetScoreAsync 暗号化キーが不正
DWC_RNK_ERROR_GET_NOMEMORY DWC_RnkGetScoreAsync メモリー不足
DWC_RNK_ERROR_CANCEL_NOTASK DWC_RnkCancelProcess 非同期処理中の処理なし
DWC_RNK_PROCESS_NOTASK DWC_RnkProcess 非同期処理中の処理なし
DWC_RNK_ERROR_INVALID_MODE 不正なモード
DWC_RNK_ERROR_NOTCOMPLETED 通信未完了
DWC_RNK_ERROR_EMPTY_RESPONSE 空のレスポンス
