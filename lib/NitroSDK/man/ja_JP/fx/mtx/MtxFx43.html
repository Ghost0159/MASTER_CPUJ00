定義

#include <nitro/fx/fx.h>

typedef union
{
    struct
    {
        fx32 _00, _01, _02;
        fx32 _10, _11, _12;
        fx32 _20, _21, _22;
        fx32 _30, _31, _32;
    };
    fx32 m[4][3];
    fx32 a[12];
}
MtxFx43;
